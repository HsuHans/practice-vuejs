Change data with method :

<button v-on:click="addOneTimes">Add 1 By Registered Method</button>

The button above has been clicked {{ counterByMethod }} times.

addOneTimes: function (event) {
   alert('You have already clicked me about ' + this.counterByMethod + ' times!')
   this.counterByMethod ++;
}
